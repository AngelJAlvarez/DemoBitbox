1. List of the cheapest item per suppliers.
SELECT  SUPPLIER.NAME, MIN(ITEM.PRICE) FROM ITEM
INNER JOIN ITEM_SUPPLIERS ON ITEM.ID  = ITEM_SUPPLIERS. ITEM_ID 
INNER JOIN SUPPLIER ON ITEM_SUPPLIERS.SUPPLIERS_ID  =  SUPPLIER.ID 
GROUP BY SUPPLIER.NAME

2. List of suppliers associated to items whose price has been reduced.
SELECT  SUPPLIER.NAME FROM ITEM
RIGHT  JOIN ITEM_SUPPLIERS ON ITEM.ID  = ITEM_SUPPLIERS. ITEM_ID 
RIGHT  JOIN SUPPLIER ON ITEM_SUPPLIERS.SUPPLIERS_ID  =  SUPPLIER.ID 
INNER JOIN ITEM_PRICE_REDUCTION ON ITEM.ID  = ITEM_PRICE_REDUCTION. ITEM_ID 
INNER JOIN PRICE_REDUCTION ON  ITEM_PRICE_REDUCTION. ITEM_ID =  PRICE_REDUCTION.ID 
GROUP BY SUPPLIER.NAME
